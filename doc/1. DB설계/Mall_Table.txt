CREATE TABLE MALL_BRAND (
	BRAND_NO			INT				NOT NULL				COMMENT '브랜드 순번'		,
	BRAND_KO_NAME		VARCHAR(30)		NOT NULL				COMMENT '한국 브랜드 이름'	,
	BRAND_KO_CONTENT	BLOB			NOT NULL				COMMENT '한국 브랜드 내용'	,
	BRAND_EN_NAME		VARCHAR(30)		NOT NULL				COMMENT '영문 브랜드 이름'	,
	BRAND_EN_CONTENT	BLOB			NOT NULL				COMMENT '영문 브랜드 내용'	,
	FILE_NO				INT				NOT NULL				COMMENT '파일번호'		,
	FILE_CODE			VARCHAR(20)		NOT NULL				COMMENT '파일코드'		,
	DEL_YN				VARCHAR(1)		NOT NULL DEFAULT 'N'	COMMENT '삭제여부'		,
	REG_DATE			DATETIME		NOT NULL DEFAULT  NOW()	COMMENT '최초등록일'		,
	MOD_DATE			DATETIME		NOT NULL DEFAULT  NOW()	COMMENT '최종수정일'		,
	PRIMARY KEY (BRAND_NO)
) COMMENT '브랜드테이블';
